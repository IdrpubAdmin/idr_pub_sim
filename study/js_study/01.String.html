문자열(String)

개행

문자열에서 줄바꿈을하는 방법

var string = 'hello!\nI\'m Seop';;

/*

hello!

I'm Seop

*/

콘솔에서 줄을 바꾸고 싶을 때 주로 \n을 사용합니다

문자열.length

문자열의 길이를 알려주는 속성

var string = "hellow";

string.length; // 5

문자열.charAt(자릿수)

문자열의 특정 자릿수의 문자를 알려줍니다

var string = "hellow";

string.charAt(1); // e

string[1]; // e

문자열.split(구분자)

구분자를 기준으로 문자열을 나누어 새 배열로 만들어줍니다

var string = "hellow";

string.split(''); // ['h', 'e', 'l', 'l', 'o']

var string2 = "cat,dog,pig";

string2.split(','); // ['cat', 'dog', 'pig']

문자열.concat(문자열)

두 문자열을 합쳐 새로문 문자열을 반환하는 함수

var string = "hellow";

var string2 = "world";

var string3 = string.concat(string2); // "hello world"
